function StudentList() {
  const { students, isLoading, error } = useStudents();
  
  // Loading state
  if (isLoading) {
    return <LoadingSpinner message="Loading students..." />;
  }
  
  // Error state
  if (error) {
    return (
      <ErrorMessage 
        title="Failed to Load Students"
        message={error.message}
        action={<Button onClick={retry}>Try Again</Button>}
      />
    );
  }
  
  // Empty state
  if (students.length === 0) {
    return (
      <EmptyState
        icon={<StudentIcon />}
        title="No Students Yet"
        description="Add your first student to get started with assessments and interventions."
        action={<Button onClick={openAddStudent}>Add Student</Button>}
      />
    );
  }
  
  // Success state
  return (
    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
      {students.map(student => (
        <StudentCard key={student.id} student={student} />
      ))}
    </div>
  );
}

**3. Validate Everything**
```typescript
// ❌ BAD
async function createStudent(data) {
  return await prisma.student.create({ data });
}

// ✅ GOOD
/**
 * Create a new student record with comprehensive validation
 */
async function createStudent(data: CreateStudentInput): Promise<Student> {
  // Validate required fields
  if (!data.firstName?.trim()) {
    throw new ValidationError('First name is required');
  }
  
  if (!data.lastName?.trim()) {
    throw new ValidationError('Last name is required');
  }
  
  if (!data.dateOfBirth) {
    throw new ValidationError('Date of birth is required');
  }
  
  // Validate date of birth is in the past
  if (new Date(data.dateOfBirth) > new Date()) {
    throw new ValidationError('Date of birth cannot be in the future');
  }
  
  // Validate tenant access
  if (!data.tenantId) {
    throw new ValidationError('Tenant ID is required');
  }
  
  // Check if student already exists
  const existingStudent = await prisma.student.findFirst({
    where: {
      firstName: data.firstName,
      lastName: data.lastName,
      dateOfBirth: data.dateOfBirth,
      tenantId: data.tenantId,
    },
  });
  
  if (existingStudent) {
    throw new ValidationError(
      'A student with this name and date of birth already exists'
    );
  }
  
  // Create student with audit logging
  try {
    const student = await prisma.student.create({
      data: {
        ...data,
        createdAt: new Date(),
        updatedAt: new Date(),
      },
    });
    
    // Log creation for audit trail
    await prisma.auditLog.create({
      data: {
        action: 'STUDENT_CREATED',
        entityType: 'STUDENT',
        entityId: student.id,
        userId: data.createdBy,
        tenantId: data.tenantId,
        timestamp: new Date(),
      },
    });
    
    return student;
  } catch (error) {
    // Log error and rethrow
    logger.error('Failed to create student', { error, data });
    throw new DatabaseError('Failed to create student');
  }
}
```

function StudentList({ students }) {
  // Fetch all assessments in single query
  const studentIds = students.map(s => s.id);
  const { data: assessments } = useQuery({
    queryKey: ['assessments', studentIds],
    queryFn: () => fetchAssessmentsBatch(studentIds),
  });
  
  // Group assessments by student
  const assessmentsByStudent = groupBy(assessments, 'studentId');
  
  return students.map(student => (
    <StudentCard 
      key={student.id}
      student={student} 
      assessments={assessmentsByStudent[student.id] || []} 
    />
  ));
}
